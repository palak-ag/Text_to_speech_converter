import cv2

 

cam = cv2.VideoCapture(0)

 

cv2.namedWindow("test")

 

img_counter = 0

 

while True:

    ret, frame = cam.read()

    cv2.imshow("test", frame)

    if not ret:

        break

    k = cv2.waitKey(1)

 

    if k%256 == 27:

        print("Escape hit, closing...")

        break

       

    elif k%256 == 32:

        img = "opencv_frame_{}.png".format(img_counter)

        cv2.imwrite(img, frame)

        print("{} written!".format(img))

        img_counter += 1

 

cam.release()

 

cv2.destroyAllWindows()
